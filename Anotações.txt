MapGet - read

// MapGet - metodo http - metodo de pegar informações, nesse caso, quando o usuário acessar a raiz do site ("/"), ele vai receber a resposta "Hello World!".

STATUS SERVER

200 - OK -> deu certo
201 - CREATED -> recurso criado. 
400 - Bad Request -> erro do cliente, enviou dados errados
202 - NOT FOUND -> Não encontrado
500 - INTERNAL SERVER ERROR -> problema no servidor


Exemplos Metodos MAP


app.MapGet("/", () => "Bem vindo ao meu site!"); 

app.MapGet("/autores", () =>
{
    List<string> nomes = new List<string>();
    {
        nomes.Add("João");
        nomes.Add("Maria");
        nomes.Add("Pedro");
    };

    return nomes;
});
//POST - criar um novo autor - passa o autor na URL, por exemplo: http://localhost:5006/autores?nome=Kafka

app.MapPost("/autores", (string nome) =>
{
    return Results.Ok($"Autor {nome} adicionado com sucesso!");
});

//PUT - atualizar um autor existente - passa o id do autor e o novo nome na URL, por exemplo: http://localhost:5006/autores/1?nome=Kafka

app.MapPut("/autores/{id}", (int id, string nome) =>
{
    return Results.Ok($"Autor com id {id} atualizado para {nome} com sucesso!");
});

//PATCH - atualizar parcialmente um autor existente - passa o id do autor e o novo nome na URL, por exemplo: http://localhost:5006/autores/1?nome=Kafka

app.MapPatch("/autores/{id}", (int id, string nome) =>
{
    return Results.Ok($"Autor com id {id} atualizado para {nome} com sucesso!");
});

//DELETE - excluir um autor existente - passa o id do autor na URL, por exemplo: http://localhost:5006/autores/1

app.MapDelete("/autores/{id}", (int id) =>
{
    return Results.Ok($"Autor com id {id} excluído com sucesso!");
});
